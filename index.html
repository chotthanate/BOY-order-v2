async function saveAndSend() { const btn = document.getElementById('btn-cloud-send'); btn.innerText = "กำลังบันทึก..."; // ข้อมูลที่จะส่ง const payload = { action: 'order', qtyData: orderCart }; try { // 1. บันทึกลง Firebase (Cloud) เหมือนเดิม const dateStr = new Date().toLocaleDateString('en-CA'); await window.setDoc(window.doc(window.db, "daily_orders", dateStr), { cart: orderCart, timestamp: new Date().toISOString() }); // 2. ยิงเข้า Google Apps Script (Webhook) // เพิ่ม ?nocache เพื่อบังคับให้ Browser ส่งข้อมูลใหม่เสมอ const finalUrl = WEBHOOK_URL + "?t=" + new Date().getTime(); await fetch(finalUrl, { method: 'POST', mode: 'no-cors', // สำคัญมากสำหรับ Google Apps Script cache: 'no-cache', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(payload) }); alert("✅ บันทึกสั่งของและแจ้งเตือน LINE เรียบร้อย!"); appState.orderLocked = true; appState.returnLocked = true; saveAppState(); updateLocksUI(); switchTab('receive'); btn.innerText = "💾 ส่งข้อมูลและสั่งของ (Cloud)"; closeCart(); } catch (e) { console.error(e); alert("เกิดข้อผิดพลาด: " + e.message); btn.innerText = "💾 ส่งข้อมูลและสั่งของ (Cloud)"; } }
